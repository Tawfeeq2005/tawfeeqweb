/* --- Safe dropdowns: attach toggle handlers to all .dropdown .head elements --- */
(() => {
  document.querySelectorAll('.dropdown .head').forEach(head => {
    head.addEventListener('click', () => {
      const container = head.closest('.dropdown');
      if (!container) return; // safety
      container.classList.toggle('open');
    });
  });
})();

/* --- Modal for certificates --- */
function openModal(id){

/* --- Scroll reveal using IntersectionObserver --- */
(() => {
  const obsOptions = {threshold:0.12};
  const observer = new IntersectionObserver((entries, obs) => {
    entries.forEach(entry => {
      if(entry.isIntersecting){
        entry.target.classList.add('fade-in');
        obs.unobserve(entry.target);
      }
    });
  }, obsOptions);

  // observe relevant elements
  document.querySelectorAll('.skill, .exp-card, .dropdown').forEach(el => observer.observe(el));
})();

/* --- Prevent dropdown errors: fallback if old inline toggleDropdown(id) used elsewhere --- */
function toggleDropdown(id){
  try{
    const el = document.getElementById(id);
    if(el){
      // if the provided id refers to a content element, toggle parent
      const parent = el.closest('.dropdown') || el;
      parent.classList.toggle('open');
    } else {
      console.warn('toggleDropdown: element with id', id, 'not found');
    }
  }catch(e){
    console.error('toggleDropdown error', e);
  }
}
